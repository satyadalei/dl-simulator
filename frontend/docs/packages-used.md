# Install Three.js, React Three Fiber, and Helper utilities
pnpm add three @react-three/fiber @react-three/drei

# Install Rapier for high-performance 3D Physics 
pnpm add @react-three/rapier

# Install Zustand for fast, lightweight game state management
pnpm add zustand

# Install types for development as a devDependency
pnpm add -D @types/three

# Launch the development server
pnpm dev
